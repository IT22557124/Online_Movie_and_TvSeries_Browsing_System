CREATE TABLE `webproject`.`user` (
  `userid` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(45) NULL,
  `password` VARCHAR(45) NULL,
  `email` VARCHAR(45) NULL,
  `contactNo` VARCHAR(45) NULL,
  `role` VARCHAR(45) NULL DEFAULT 'user',
  PRIMARY KEY (`userid`));


CREATE TABLE `webproject`.`celebrity` (
  `celebrityId` INT NOT NULL AUTO_INCREMENT,
  `celebrityName` VARCHAR(45) NULL,
  `biography` LONGTEXT NULL,
  `dob` DATE NULL,
  `gender` VARCHAR(45) NULL,
  `country` VARCHAR(45) NULL,
  `posterImage` VARCHAR(45) NULL,
  PRIMARY KEY (`celebrityId`));

CREATE TABLE `webproject`.`movies` (
  `movieid` INT NOT NULL AUTO_INCREMENT,
  `movieName` VARCHAR(45) NULL,
  `description` LONGTEXT NULL,
  `director` VARCHAR(45) NULL,
  `country` VARCHAR(45) NULL,
  `releaseYear` DATE NULL,
  `mainActors` VARCHAR(45) NULL,
  `budget` VARCHAR(45) NULL,
  `rating` VARCHAR(45) NULL,
  `posterImage` VARCHAR(45) NULL,
  `trailer` VARCHAR(45) NULL,
  PRIMARY KEY (`movieid`));


CREATE TABLE `webproject`.`tvseries` (
  `tvSeriesId` INT NOT NULL AUTO_INCREMENT,
  `tvSeriesName` VARCHAR(45) NULL,
  `description` VARCHAR(45) NULL,
  `director` VARCHAR(45) NULL,
  `country` VARCHAR(45) NULL,
  `releaseYear` VARCHAR(45) NULL,
  `mainActors` VARCHAR(45) NULL,
  `budget` VARCHAR(45) NULL,
  `ratings` VARCHAR(45) NULL,
  `posterImage` VARCHAR(45) NULL,
  `trailer` VARCHAR(45) NULL,
  `seasons` VARCHAR(45) NULL,
  PRIMARY KEY (`tvSeriesId`));
